config = require('../config/config');
const pool = require('../config/database')
module.exports.authenticate = (email) => {

    return new Promise((resolve, reject) => {
        pool.getConnection((err, connection) => {
            if (err) {
                reject(err);

            } else {
                try {
                    connection.query(`SELECT user.user_id, fullname, email, user_password, role_name, user.role_id  
                   FROM user INNER JOIN role ON user.role_id=role.role_id AND email='${email}'`, {}, (error, rows) => {
                        if (error) {
                            reject(err);

                        } else {
                            if (rows.length == 1) {
                                console.log(rows);
                                resolve(rows);

                            } else {
                                reject('login has failed');
                                // return callback('Login has failed', null);
                            }
                        }
                        connection.release();

                    });
                } catch (error) {
                    reject(error);
                    // return callback(error, null);
                }
            }
        }); //End of getConnection
    });

} //End of authenticate
